Montgomery CO. Covid-19 Status.

This notebook extracts data from the Montgomery Co. Texas health department web site: https://mcphd-tx.org/coronavirus-covid-19/confirmed-cases/ and creates several plots to visualize the data.

# Import needed modules
from IPython.display import Markdown as md
import pandas as pd
import datetime,re
from scipy import stats
from numpy import *
from matplotlib.pyplot import *
from pandas.plotting import register_matplotlib_converters
register_matplotlib_converters()

url1='https://mcphd-tx.org/coronavirus-covid-19/confirmed-cases/'
dat_a=pd.read_html(url1)[0]
dat_b=pd.read_html(url1)[1]
dat_b.columns=dat_a.columns
dat=dat_a.append(dat_b)
clms=list(dat.columns)
clms[0]='N'
dat.columns=clms
dat.DATE=pd.to_datetime(dat.DATE)
dat['days']=(dat.DATE-dat.DATE.iloc[0]).dt.days
agg_fnc={'N':'count','days':'first'}
#
dat1=dat.groupby(by='DATE').agg(agg_fnc)
dat1['cases']=dat1.N.cumsum()

pcti=dat1.cases[1:].pct_change().mean()
#
dbltm=log(2)/log(1+pcti)
#
dat.loc[dat.STATUS=='Under investigation','STATUS']='Under Investigation'
#
D=dat[dat.STATUS=='Deceased'].groupby(by='DATE').agg(agg_fnc)
D.columns=['D','days']
#
Rec=dat[dat.STATUS=='Recovered'].groupby(by='DATE').agg(agg_fnc)
Rec.columns=['R','days']
#
AH=dat[dat.STATUS=='At Home'].groupby(by='DATE').agg(agg_fnc)
AH.columns=['H','days']
#
AHR=dat[dat.STATUS=='At Home (released from hospital)'].groupby(by='DATE').agg(agg_fnc)
AHR.columns=['HR','days']
#
UI=dat[dat.STATUS=='Under Investigation'].groupby(by='DATE').agg(agg_fnc)
UI.columns=['UI','days']

dat2=dat1.merge(D.D,how='left',left_on='DATE',right_on='DATE')
dat2.loc[pd.isnull(dat2.D),'D']=0
dat2=dat2.merge(Rec.R,how='left',left_on='DATE',right_on='DATE')
dat2.loc[pd.isnull(dat2.R),'R']=0
dat2=dat2.merge(AH.H,how='left',left_on='DATE',right_on='DATE')
dat2.loc[pd.isnull(dat2.H),'H']=0
dat2=dat2.merge(AHR.HR,how='left',left_on='DATE',right_on='DATE')
dat2.loc[pd.isnull(dat2.HR),'HR']=0
dat2=dat2.merge(UI.UI,how='left',left_on='DATE',right_on='DATE')
dat2.loc[pd.isnull(dat2.UI),'UI']=0
dat2['A']=dat2.H+dat2.HR+dat2.UI
dat2['INAC']=dat2.D+dat2.R
#
dat2['active']=dat2.A.cumsum()
dat2['deceased']=dat2.D.cumsum()
dat2['hospitalized']=dat2.H.cumsum()
dat2['recovered']=dat2.R.cumsum()
dat2['ui']=dat2.UI.cumsum()
#
dat2['pcti']=dat2.cases[1:].pct_change()
dat2['pcti_ma5']=dat2.pcti.rolling(window=5).mean()
dat2['dbltm_ma5']=log(2)/log(1+dat2.pcti_ma5)
inc_today=dat2.N.iloc[-1]
lastD=dat2.index[-1]
dths=dat2.deceased.iloc[-1]
#
# Write statistics as markdown
md(('### Montgomery Co. TX Statistics for {}: ###\n&nbsp;&nbsp;   Total Cases: {}&nbsp;&nbsp;&nbsp;&nbsp; Active Cases: {}'+\
   '&nbsp;&nbsp;&nbsp;&nbsp; Daily Increase: {}'+\
   '&nbsp;&nbsp;&nbsp;&nbsp; Case doubling time (days): {:.2f}\n\n'+\
   '&nbsp;&nbsp; Recovered: {}&nbsp;&nbsp;&nbsp;&nbsp; Hospitalized {}&nbsp;&nbsp;&nbsp;&nbsp; Deaths: {}'+\
   '&nbsp;&nbsp;&nbsp;&nbsp; Under Investigation {}').format(\
    dat2.index[-1].strftime('%m/%d/%Y'),dat2.cases.iloc[-1],dat2.active.iloc[-1],dat2.N.iloc[-1],dat2.dbltm_ma5.iloc[-1],\
    dat2.recovered.iloc[-1],dat2.hospitalized.iloc[-1],dat2.deceased.iloc[-1],dat2.ui.iloc[-1]))

figure(figsize=[8,6])

plot(dat2.index,dat2.deceased,color='blue',marker='*',linestyle='-',label='Deceased')
plot(dat2.index,dat2.active,color='orange',marker='*',linestyle='-',label='Active')
plot(dat2.index,dat2.recovered,color='green',marker='*',linestyle='-',label='Recovered')
plot(dat2.index,dat2.hospitalized,color='red',marker='*',linestyle='-',label='Hospitalized')
grid('both')
legend()
t=title('Covid-19 Cases %s\nMontgomery Co. TX' % (dat2.index[-1].strftime('%m/%d/%Y')))

figure(figsize=[10,6])
plot(dat2.index,dat2.dbltm_ma5)
grid('both')
ylabel('Time to Double (days)')
t=title('Covid-19 Time to Double Number of Cases (5day moving average) %s\nMontgomery Co. TX' % (dat2.index[-1].strftime('%m/%d/%Y')))
